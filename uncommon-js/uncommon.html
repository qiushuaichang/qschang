var arr = ['dog','cat','hen']
        arr.toString()          // 'dog,cat,hen'
        arr.concat('mouse')     // ['dog','cat','hen','mouse']
        arr.pop()               // 'hen'
        arr.shift()             // 'dog'
        arr.splice(1,2)         // ['cat','hen']
    
        js 中 in 运算符
        var obj = {a:"1",b:"2"}     "a" in obj      //true,在对象中，in前边的参数"a"代表obj的key
        var arr = [a,21,"str",33]   2 in arr        //true,在对象中，in前边的参数代表arr的索引号
                                    21 in  arr      //false
    
        str = 'hello happy world!'
        str.slice()         str.slice(6)        //'happy world!'  ==>slice(6,str.length-1)  含头不含尾
        str.substring()     //同上
        str.substr()        str.substr(-3,3)    //'ld!'     -3代表str.length+(-3);  3代表向后截取几位
                            str.substr(8,3)     //'ppy'     
    
        判断是否是数组
        arr = [1,2,3]
        alert(arr instanceof Array)     //true
        alert(arr.constructor === Array)    //true       同样的方法可以判断是否为对象，而且此方法更准确
